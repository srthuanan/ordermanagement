    function fetchLiveLocation(vin) {
        return new Promise(function(resolve){

            req.setRequestHeader("OData-Version", "4.0");
            req.setRequestHeader("Accept", "application/json");
            req.setRequestHeader("Content-Type", "application/json; charset=utf-8");

                                updateReq.setRequestHeader("apikey", apiKey);

                                updateReq.setRequestHeader("Content-Type", "application/json");
                                updateReq.send(JSON.stringify({ extension_reason: "GPS:" + data.Lat + "," + data.Long }));
                                
                                resolve({ vin: vin, lat: data.Lat, lng: data.Long, success: true });
                            }
                            else resolve({ vin: vin, success: false });
                        } catch(e) { resolve({ vin: vin, success: false }); }
                    } else resolve({ vin: vin, success: false });
                }
            };

        });
    }
