    function fetchDMSMetadataBatch(vins, index, onComplete) {
        if (index >= vins.length) { onComplete(); return; }
        var batchSize = 15;
        var batch = vins.slice(index, index + batchSize);
        var filter = batch.map(function(v){ return "xts_chassisnumber eq '" + v + "'"; }).join(' or ');

        function fetchSingleBatchPage(query) {
            Xrm.WebApi.retrieveMultipleRecords("xts_inventorynewvehicle", query).then(function(result) {
                result.entities.forEach(function(item) {
                    var vin = String(item.xts_chassisnumber || '').trim().toUpperCase();
                    if(vin && !dmsMetadataMap[vin]) {
                        dmsMetadataMap[vin] = {
                            model: item.xts_productdescription,
                            color: item["_xts_vehicleexteriorcolorid_value@OData.Community.Display.V1.FormattedValue"],
                            interior: item["_xts_vehicleinteriorcolorid_value@OData.Community.Display.V1.FormattedValue"],
                            site: item["_xts_siteid_value@OData.Community.Display.V1.FormattedValue"]
                        };
                    }
                });
                var nextLink = result["@odata.nextLink"] || result.nextLink;
                if (nextLink) fetchSingleBatchPage(nextLink.substring(nextLink.indexOf('?')));
                else fetchDMSMetadataBatch(vins, index + batchSize, onComplete);
            }, function() { fetchDMSMetadataBatch(vins, index + batchSize, onComplete); });
        }
        var query = "?$filter=" + filter;
        fetchSingleBatchPage(query);
    }

    function fetchLiveLocation(vin) {
        return new Promise(function(resolve){
            var payload = { itv_requestObject: JSON.stringify({ data: [{ vinCode: vin }], isUpdate: false }) };
            var req = new XMLHttpRequest();
            req.open("POST", "/api/data/v9.2/itv_trackingvehicleposition", true);
            req.setRequestHeader("OData-MaxVersion", "4.0");
            req.setRequestHeader("OData-Version", "4.0");
            req.setRequestHeader("Accept", "application/json");
            req.setRequestHeader("Content-Type", "application/json; charset=utf-8");
            req.onreadystatechange = function() {
                if (this.readyState === 4) {
                    if (this.status === 200 || this.status === 204) {
                        try {
                            var data = JSON.parse(JSON.parse(this.response).itv_responseObject);
                            if(data && data.Lat && data.Long) {
                                var speedVal = data.Speed || data.speed || data.Velocity || data.velocity || 0;
                                var headingVal = data.Heading || data.heading || data.Bearing || data.bearing || data.Angle || data.angle || 0;
                                
                                var updateReq = new XMLHttpRequest();
                                updateReq.open("POST", "https://jwvgxqrkjlbewvpkvucj.supabase.co/rest/v1/car_telemetry", true);
                                updateReq.setRequestHeader("apikey", apiKey);
                                updateReq.setRequestHeader("Authorization", "Bearer " + apiKey);
                                updateReq.setRequestHeader("Content-Type", "application/json");
                                updateReq.setRequestHeader("Prefer", "resolution=merge-duplicates");
                                updateReq.send(JSON.stringify({ 
                                    vin: vin, 
                                    lat: parseFloat(data.Lat), 
                                    lng: parseFloat(data.Long),
                                    speed: parseFloat(speedVal),
                                    heading: parseFloat(headingVal),
                                    updated_at: new Date().toISOString()
                                }));
                                resolve({ vin: vin, lat: data.Lat, lng: data.Long, speed: speedVal, heading: headingVal, success: true });
                            }
                            else resolve({ vin: vin, success: false });
                        } catch(e) { resolve({ vin: vin, success: false }); }
                    } else resolve({ vin: vin, success: false });
                }
            };
            req.send(JSON.stringify(payload));
        });
    }

    async function fetchLiveLocations(vins) {
        var results = [];
        var concurrency = 5;
        for (var i = 0; i < vins.length; i += concurrency) {
            var batch = vins.slice(i, i + concurrency);
            results = results.concat(await Promise.all(batch.map(function(v){ return fetchLiveLocation(v); })));
        }
        return results;
    }
